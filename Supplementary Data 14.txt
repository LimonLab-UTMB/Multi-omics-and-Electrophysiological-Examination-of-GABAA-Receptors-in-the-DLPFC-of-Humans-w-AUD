#Title: Converting Cliff's delta to Cohen's d - multi-omic data
#Description: This code was used to calculate a Cliff's delta values of our multi-omic data to a Cohen's d value that can be used in a power analysis
# R version used: 4.4.1



#1. Define the Conversion Function
convert_cliff_delta_to_cohens_d = function(cliff_delta_value) {
  # Basic check to ensure the input is within the valid range for Cliff's Delta
  if (cliff_delta_value < -1 || cliff_delta_value > 1) {
    stop("Error: Cliff's Delta must be between -1 and 1.")
  }
  
  # The core conversion formula ; qnorm is thw quantile function of the normal distribution
  cohens_d_result = sqrt(2) * qnorm((cliff_delta_value + 1) / 2)
  
  return(cohens_d_result)
}
